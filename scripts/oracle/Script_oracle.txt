-- =========================
CREATE TABLE clients ( 
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  doc_type VARCHAR2(20) NOT NULL,
  doc_number VARCHAR2(30) UNIQUE NOT NULL,
  full_name VARCHAR2(100) NOT NULL,
  phone VARCHAR2(20),
  email VARCHAR2(100),
  status VARCHAR2(20) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================
CREATE TABLE vehicles (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  client_id NUMBER NOT NULL,
  plate VARCHAR2(20) UNIQUE NOT NULL,
  brand VARCHAR2(100),
  description VARCHAR2(200),
  status VARCHAR2(20) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_client_vehicles FOREIGN KEY (client_id) REFERENCES clients(id)
);

-- =========================
CREATE TABLE rates (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name VARCHAR2(100) NOT NULL,
  amount_hour NUMBER NOT NULL,
  status VARCHAR2(20) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================
CREATE TABLE employees (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  full_name VARCHAR2(100) NOT NULL,
  role VARCHAR2(50) NOT NULL,
  status VARCHAR2(20) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================
CREATE TABLE shifts (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  employee_id NUMBER NOT NULL,
  start_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  end_time TIMESTAMP,
  status VARCHAR2(20) DEFAULT 'open',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_employee_shifts FOREIGN KEY (employee_id) REFERENCES employees(id)
);

-- =========================
CREATE TABLE entries (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  vehicle_id NUMBER NOT NULL,
  shift_id NUMBER NOT NULL,
  entry_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  exit_time TIMESTAMP,
  status VARCHAR2(20) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_vehicle_entries FOREIGN KEY (vehicle_id) REFERENCES vehicles(id),
  CONSTRAINT fk_shift_entries FOREIGN KEY (shift_id) REFERENCES shifts(id)
);

-- =========================
CREATE TABLE tickets (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  entry_id NUMBER UNIQUE,
  name VARCHAR2(100) NOT NULL,
  status VARCHAR2(20) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_entry_tickets FOREIGN KEY (entry_id) REFERENCES entries(id)
);

-- =========================
CREATE TABLE payments (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  ticket_id NUMBER NOT NULL,
  rate_id NUMBER NOT NULL,
  method VARCHAR2(50),
  amount NUMBER,
  pay_date DATE,
  reference VARCHAR2(100),
  status VARCHAR2(20) DEFAULT 'completed',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_ticket_payments FOREIGN KEY (ticket_id) REFERENCES tickets(id),
  CONSTRAINT fk_rate_payments FOREIGN KEY (rate_id) REFERENCES rates(id)
);
